DatabaseOptimizer
- log_slow_query() — Track slow queries with context

- recommend_indexes() — Suggest indexes with ROI estimates

- suggest_materialized_view() — Recommend views for expensive queries

CacheLayer (Main interface)
- get() — Get with Redis fallback to local
- set() — Set with optional Redis only
- delete() — Delete from both tiers
- cache_decorator() — Function result caching
- cache_kpi() — Predefined KPI cache
- cache_market_data() — Predefined market data cache
- cache_agent_status() — Predefined agent status cache
- cache_query_result() — Predefined query result cache
- get_stats() — Cache hit rate and metrics

AsyncJobQueue
- enqueue() → Job — Queue task with priority
- execute_job() — Run single job with timeout
- start_workers() — Start async worker pool
- stop_workers() — Shutdown workers gracefully
- get_job_status() — Check job progress
- get_queue_status() — Queue depth and active jobs
- cancel_job() — Cancel pending job
- get_stats() — Success rate, execution time metrics

PerformanceMonitor
- record_metric() — Log any metric
- record_query_time() — Log database queries
- record_endpoint_time() — Log HTTP endpoints
- get_metric_stats() — Calculate p95, p99, stdev
- detect_bottlenecks() — Find performance issues
- get_system_health() — Overall health status
- get_performance_report() — Comprehensive report
- get_alerts() — Query performance alerts

LoadTestRunner
- run_load_test() → LoadTestResult — Execute load test
- get_test_results() — Results for specific test
- get_all_results() — All historical results
- compare_results() — A/B comparison between tests

QUICK USAGE EXAMPLES

# 1. Database Optimization
from app.core.performance import DatabaseOptimizer

optimizer = DatabaseOptimizer()
await optimizer.log_slow_query("SELECT...", 250, 50000, 432)
report = optimizer.get_slow_query_report(limit=50)

# 2. Caching
from app.core.performance import CacheLayer

cache = CacheLayer(redis_client=redis)
await cache.cache_kpi("agent_001", "conversion_rate", 0.087, ttl_seconds=1800)
rate = await cache.get_kpi("agent_001", "conversion_rate")

# 3. Async Jobs
from app.core.performance import AsyncJobQueue, JobPriority

queue = AsyncJobQueue(worker_count=4)
job = queue.enqueue(task=async_import, name="import_agents", priority=JobPriority.HIGH)

# 4. Performance Monitoring
from app.core.performance import PerformanceMonitor

monitor = PerformanceMonitor()
monitor.record_query_time("fetch_users", 45.3, rows_affected=100)
health = monitor.get_system_health()
report = monitor.get_performance_report()

# 5. Load Testing
from app.core.performance import LoadTestRunner, LoadTestConfig

runner = LoadTestRunner()
config = LoadTestConfig(name="peak_load", target_function=api_handler, concurrent_users=1000)
result = await runner.run_load_test(config, request_generator)
